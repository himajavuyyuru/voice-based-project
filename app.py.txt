import streamlit as st
import matplotlib.pyplot as plt
import numpy as np
from reportlab.pdfgen import canvas
import io

# Page configuration
st.set_page_config(
    page_title="Voice Based Concept Understanding Analyser",
    page_icon="🎤",
    layout="wide"
)

# Initialize session state
if "transcript" not in st.session_state:
    st.session_state.transcript = ""

if "score" not in st.session_state:
    st.session_state.score = 0

if "evaluation" not in st.session_state:
    st.session_state.evaluation = ""

if "filler" not in st.session_state:
    st.session_state.filler = 0

if "confidence" not in st.session_state:
    st.session_state.confidence = 0

if "analysis_done" not in st.session_state:
    st.session_state.analysis_done = False


# Function to validate audio
def validate_audio(uploaded_file):
    allowed_extensions = ["wav", "mp3", "m4a"]

    extension = uploaded_file.name.split(".")[-1].lower()

    if extension not in allowed_extensions:
        return False, "Unsupported file format."

    if uploaded_file.size == 0:
        return False, "Corrupted or empty audio file."

    return True, "Valid"


# Dummy analysis function
def analyze_audio(file):

    transcript = """
    This is a sample transcript generated from the uploaded audio.
    """

    score = 88

    if score >= 80:
        evaluation = "Excellent Concept Understanding"
    elif score >= 60:
        evaluation = "Good Concept Understanding"
    else:
        evaluation = "Needs Improvement"

    return transcript, score, evaluation


# PDF generator
def create_pdf():

    buffer = io.BytesIO()

    c = canvas.Canvas(buffer)

    c.drawString(
        100,
        800,
        "Voice Based Concept Understanding Report"
    )

    c.drawString(
        100,
        760,
        f"Concept Score: {st.session_state.score}%"
    )

    c.drawString(
        100,
        740,
        f"Filler Ratio: {st.session_state.filler}%"
    )

    c.drawString(
        100,
        720,
        f"Confidence: {st.session_state.confidence}%"
    )

    c.drawString(
        100,
        700,
        st.session_state.evaluation
    )

    c.save()

    buffer.seek(0)

    return buffer


# Title
st.title("🎤 Voice Based Concept Understanding Analyser")

st.write(
    "Upload an audio file to analyze concept understanding and evaluate performance."
)

# Sidebar
st.sidebar.header("Instructions")
st.sidebar.write("1. Upload an audio file")
st.sidebar.write("2. Click Analyze")
st.sidebar.write("3. View transcript and evaluation")

# File uploader
uploaded_file = st.file_uploader(
    "Upload Audio File",
    type=["wav", "mp3", "m4a"]
)

if uploaded_file:

    valid, message = validate_audio(uploaded_file)

    if not valid:
        st.error(message)

    else:

        st.success("Audio uploaded successfully!")

        st.audio(uploaded_file)

        st.subheader("File Details")
        st.write(f"Filename: {uploaded_file.name}")
        st.write(f"Size: {uploaded_file.size/1024:.2f} KB")

        if st.button("Analyze Concept Understanding"):

            with st.spinner("Processing audio..."):

                transcript, score, evaluation = analyze_audio(
                    uploaded_file
                )

                st.session_state.transcript = transcript
                st.session_state.score = score
                st.session_state.evaluation = evaluation
                st.session_state.filler = 5
                st.session_state.confidence = 92
                st.session_state.analysis_done = True


# Display results
if st.session_state.analysis_done:

    st.divider()

    st.header("Analysis Results")

    col1, col2 = st.columns(2)

    with col1:
        st.subheader("Transcript")

        st.text_area(
            "Generated Transcript",
            st.session_state.transcript,
            height=200
        )

    with col2:
        st.subheader("Evaluation")

        st.metric(
            "Concept Understanding Score",
            f"{st.session_state.score}%"
        )

        if st.session_state.score >= 80:
            st.success(st.session_state.evaluation)
        elif st.session_state.score >= 60:
            st.warning(st.session_state.evaluation)
        else:
            st.error(st.session_state.evaluation)

    # Progress
    st.subheader("Performance")
    st.progress(st.session_state.score / 100)

    # Metrics
    st.subheader("Evaluation Metrics")

    col1, col2 = st.columns(2)

    with col1:
        st.metric(
            "Filler Word Ratio",
            f"{st.session_state.filler}%"
        )

    with col2:
        st.metric(
            "Confidence Score",
            f"{st.session_state.confidence}%"
        )

    # Waveform
    st.subheader("Waveform Visualization")

    x = np.linspace(0, 10, 1000)
    y = np.sin(10 * x)

    fig, ax = plt.subplots()

    ax.plot(x, y)

    st.pyplot(fig)

    # PDF Download
    pdf = create_pdf()

    st.download_button(
        label="Download PDF Report",
        data=pdf,
        file_name="evaluation_report.pdf",
        mime="application/pdf"
    )

# Footer
st.divider()

st.caption(
    "Voice Based Concept Understanding Analyser | Streamlit UI"
)
